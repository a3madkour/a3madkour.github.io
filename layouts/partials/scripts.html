{{- /* Multi-entry bundling: one core bundle on every page, plus a section-scoped bundle on /essays/ and /garden/. Hugo's js.Build can't emit code-split chunks from a single entry (esbuild's splitting requires outdir mode), so we get per-section payloads by building independent entries. Each call gets its own fingerprint + SRI. */ -}} {{- $coreOpts := dict "targetPath" "js/core.js" "minify" true -}} {{- $core := resources.Get "js/index.js" | js.Build $coreOpts | fingerprint -}} {{- if eq .Section "essays" }} {{- $essayOpts := dict "targetPath" "js/essay.js" "minify" true -}} {{- $essay := resources.Get "js/entry-essay.js" | js.Build $essayOpts | fingerprint }} {{- end }} {{- if eq .Section "garden" }} {{- $gardenOpts := dict "targetPath" "js/garden.js" "minify" true -}} {{- $garden := resources.Get "js/entry-garden.js" | js.Build $gardenOpts | fingerprint }} {{- end }}
