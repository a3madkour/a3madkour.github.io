{{- /* Multi-entry bundling: one core bundle on every page; a section-scoped bundle on /essays/ and /garden/; a page-narrow bundle on /research/ (index + standalone graph page only — theme/question pages stay on core). Hugo's js.Build can't emit code-split chunks from a single entry (esbuild's splitting requires outdir mode), so we get per- payload binaries by building independent entries. Each call gets its own fingerprint + SRI. */ -}} {{- $coreOpts := dict "targetPath" "js/core.js" "minify" true -}} {{- $core := resources.Get "js/index.js" | js.Build $coreOpts | fingerprint -}} {{- if eq .Section "essays" }} {{- $essayOpts := dict "targetPath" "js/essay.js" "minify" true -}} {{- $essay := resources.Get "js/entry-essay.js" | js.Build $essayOpts | fingerprint }} {{- end }} {{- if eq .Section "garden" }} {{- $gardenOpts := dict "targetPath" "js/garden.js" "minify" true -}} {{- $garden := resources.Get "js/entry-garden.js" | js.Build $gardenOpts | fingerprint }} {{- end }} {{- $loadResearch := or (and (eq .Section "research") (eq .Kind "section")) (and (eq .Section "research") (eq .Layout "graph")) -}} {{- if $loadResearch }} {{- $researchOpts := dict "targetPath" "js/research.js" "minify" true -}} {{- $research := resources.Get "js/entry-research.js" | js.Build $researchOpts | fingerprint }} {{- end }} {{- /* Works split: umbrella + standalone graph → entry-works-umbrella.js (~110 KB w/ d3) per-item works pages → entry-works.js (~6 KB, no d3) */ -}} {{- $isUmbrella := or (eq .RelPermalink "/works/") (eq .RelPermalink "/works/graph/") -}} {{- if $isUmbrella }} {{- $worksUmbOpts := dict "targetPath" "js/works-umbrella.js" "minify" true -}} {{- $worksUmb := resources.Get "js/entry-works-umbrella.js" | js.Build $worksUmbOpts | fingerprint }} {{- else if eq .Section "works" }} {{- $worksOpts := dict "targetPath" "js/works.js" "minify" true -}} {{- $works := resources.Get "js/entry-works.js" | js.Build $worksOpts | fingerprint }} {{- end }} {{- /* Library: load entry-library.js on leaf pages only. The umbrella (/library/) has no chips and no rows — skip it. */ -}} {{- $isLibraryLeaf := and (eq .Section "library") (ne .RelPermalink "/library/") -}} {{- if $isLibraryLeaf }} {{- $libOpts := dict "targetPath" "js/library.js" "minify" true -}} {{- $lib := resources.Get "js/entry-library.js" | js.Build $libOpts | fingerprint }} {{- end }}
