{{- /* Input: .date — a time.Time (or YAML date) Output: "Nd ago" / "N days ago" / "today" / "yesterday" — short form. */ -}} {{- $now := now -}} {{- $then := .date -}} {{- $diff := math.Floor (div (sub $now.Unix $then.Unix) 86400) -}} {{- if le $diff 0 -}}today {{- else if eq $diff 1.0 -}}yesterday {{- else if lt $diff 30.0 -}}{{ int $diff }}d ago {{- else if lt $diff 365.0 -}}{{ math.Floor (div $diff 30) }}mo ago {{- else -}}{{ math.Floor (div $diff 365) }}y ago {{- end -}}
