Inputs: dimensions — slice of dicts: { key, label, values } key: machine name (used as data-dim, e.g. "tag") label: display label (e.g. "Tag") values: slice of strings (the chip values) mode — "single" (legacy, single chip active across all dims) — "and" (default, per-dim active state, AND combined) Suppression rule: dimensions with len(values) < 2 are not rendered. */ -}} {{- $mode := .mode | default "and" -}} {{- $dims := .dimensions | default slice -}} {{- $renderable := slice -}} {{- range $dims -}} {{- if ge (len .values) 2 -}} {{- $renderable = $renderable | append . -}} {{- end -}} {{- end -}} {{- if $renderable -}}
{{- range $renderable -}} {{- $key := .key -}}
{{ .label }} {{- range .values -}} {{- end -}}
{{- end -}}
